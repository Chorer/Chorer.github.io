    <div>
        <p class="1">1</p>
        <p class="2">2</p>
        <p class="3">3</p>
        <p class="4">4</p>
        <p class="5">5</p>
    </div>

<script src="https://lib.sinaapp.com/js/jquery/2.0.2/jquery-2.0.2.min.js"></script>
<script>
    var allp=$("div p");
    for(var i=0;i<allp.length;i++){
        allp[i].attr("class","i+2");
